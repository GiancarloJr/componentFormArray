Como utilizar componente:

Para utilização do componente {{ componentitle }}, importar {{ componentitle | NameModule }} adicionando no módulo destino: 

        
          import { NgModule } from '@angular/core';

          import { CommomModules } from 'projects/virtue/lib/modules/public-api';
          import { DocSectiontotalComponent } from './doc-sectiontotal.component';
          import { DocSectiontotalRoutingModule } from './doc-sectiontotal-routing.module';
          import { SectionTotalModule } from 'projects/virtue/lib/public-api';

          @NgModule({
            declarations: [
              DocSectiontotalComponent
            ],
            imports: [
              // Library
              CommomModules,
              SectionTotalModule,

              // Local
              DocSectiontotalRoutingModule
            ]
          })
          export class DocSectiontotalModule { }

Inserindo HTML para utilização do componente: 

        
                <virtue-section [section-showdivider]='false'
                                fxFlex=50
                                section-icon="fad fa-equals"
                                section-title="Total de Vendas">

                    <div fxLayout="row"
                        fxLayoutAlign="space-between stretch">

                      <!-- TOTAL ENTRADAS  -->
                      <virtue-sectiontotal fxFlex
                                          title="A - Entradas (R$)"
                                          value-color="#3c763d">
                        {{totalEntradas | number:'.2-2'}}
                      </virtue-sectiontotal>

                      <!-- TOTAL SAÍDAS  -->
                      <virtue-sectiontotal fxFlex
                                          title="B - Saídas (R$)"
                                          value-color="#a94442">
                        {{totalSaidas | number:'.2-2'}}
                      </virtue-sectiontotal>

                      <!-- SALDO  -->
                      <virtue-sectiontotal fxFlex
                                          title="(A-B) Saldo (R$)"
                                          value-color="#a94442">
                        {{(totalEntradas - totalSaidas) | number:'.2-2'}}
                      </virtue-sectiontotal>

                    </div>

                </virtue-section>
