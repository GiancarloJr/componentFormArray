Como utilizar componente:

Para utilização do componente {{ componentitle }}, importar {{ componentitle | NameModule }} adicionando no módulo destino: 

        
          import { NgModule } from '@angular/core';

          import { CommomModules } from 'projects/virtue/lib/modules/public-api';
          import { DocPdvsectionComponent } from './doc-pdvsection.component';
          import { DocPdvsectionRoutingModule } from './doc-pdvsection-routing.module';
          import { PDVSectionModule } from 'projects/virtue/lib/components/public-api';

          @NgModule({
            declarations: [
              DocPdvsectionComponent
            ],
            imports: [
              // Library
              CommomModules,
              PDVSectionModule,

              // Local
              DocPdvsectionRoutingModule
            ]
          })
          export class DocPdvsectionModule { }

Inserindo HTML para utilização do componente: 

        
          <virtue-pdvsection section-title="Teste Componente PDV Section"></virtue-pdvsection>
