Como utilizar componente:

Para utilização do componente {{ componentitle }}, importar {{ componentitle | NameModule }} adicionando no módulo destino: 

        
          import { NgModule } from '@angular/core';

          import { CommomModules } from 'projects/virtue/lib/modules/commom.modules';
          import { DocFielderrorComponent } from './doc-fielderror.component';
          import { DocFielderrorRoutingModule } from './doc-fielderror-routing.module';
          import { FieldErrorModule } from 'projects/virtue/lib/components/public-api';


          @NgModule({
            declarations: [
              DocFielderrorComponent
            ],
            imports: [

              // Library
              CommomModules,
              FieldErrorModule,

              // Local
              DocFielderrorRoutingModule
            ]
          })
          export class DocFielderrorModule { }
