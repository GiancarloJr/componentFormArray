Como utilizar componente:

Para utilização do componente {{ componentitle }}, importar {{ componentitle | NameModule }} adicionando no módulo destino: 

        
          import { NgModule } from '@angular/core';

          import { CommomModules } from 'projects/virtue/lib/modules/public-api';
          import { DocLoaderComponent } from './doc-loader.component';
          import { DocLoaderRoutingModule } from './doc-loader-routing.module';
          import { LoaderModule } from 'projects/virtue/lib/components/public-api';


          @NgModule({
            declarations: [
              DocLoaderComponent
            ],
            imports: [
              // Library
              CommomModules,
              LoaderModule,

              // Local
              DocLoaderRoutingModule
            ]
          })
          export class DocLoaderModule { }

Inserindo HTML para utilização do componente:

Exemplo {{ componentitle }} 

    
      <div class="loader-container"
              fxLayout="column"
              fxLayoutAlign="center center">
        <mat-spinner color="primary"
                        mode="indeterminate"
                        strokeWidth="5">
        </mat-spinner>
        <img width="96"
            [src]='imageUrl'>
        <h3 class="loader-message">{ { message }}</h3>
      </div>
      
    
        // Angular
        import { Component, Inject, InjectionToken, Input } from '@angular/core';

        export const LOADER_IMAGE = new InjectionToken('loaderImage');

        @Component({
          selector: 'virtue-loader',
          templateUrl: './loader.component.html',
          styleUrls: ['./loader.component.scss']
        })
        export class LoaderComponent {

          //#region Constructor

          constructor(
            @Inject(LOADER_IMAGE) private _loaderimage) {

            this.imageUrl = _loaderimage;

          }

          //#endregion

          //#region Input Properties

          @Input() imageUrl: string;
          @Input() message: string;

          //#endregion

        }
